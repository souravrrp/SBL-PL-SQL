select 
SHOP_CODE,PART_NO,
sum(QTY_ONHAND) as ONHAND from
IFSAPP.INVENTORY_PART_IN_STOCK_TAB,
IFSAPP.SHOP_DTS_INFO,
IFSAPP.PRODUCT_CATEGORY_INFO
WHERE IFSAPP.SHOP_DTS_INFO.SHOP_CODE=IFSAPP.INVENTORY_PART_IN_STOCK_TAB.CONTRACT
AND IFSAPP.INVENTORY_PART_IN_STOCK_TAB.PART_NO=IFSAPP.PRODUCT_CATEGORY_INFO.PRODUCT_CODE

and PART_NO in ('GJREF-GFE-25-SMTE-B','GJREF-GFE-25-SVTE-B','GJREF-GFE-30-CMTE-B','GJREF-GFE-30-CVTE-B','GJREF-GFE-35-LMTE-B','GJREF-GFE-35-LVTE-B')
group by SHOP_CODE,PART_NO

having sum(QTY_ONHAND)>0

order by part_no
