select
r.SALES_RETURN_NO,
r.CUSTOMER_NO,
r.CONTRACT,
l.ORDER_NO,
r.DATE_REQUESTED,
l.SALE_UNIT_PRICE 

from
IFSAPP.HPNRET_SALES_RETURN R,
IFSAPP.HPNRET_SALES_RET_LINE L
where r.SALES_RETURN_NO= L.SALES_RETURN_NO
and  r.REFUN_STMT_ID is null
and r.OBJSTATE='ReturnCompleted'
and r.HPNRET_SALES_RET_TYPE='NonHP'
and l.SALE_UNIT_PRICE >0
order by r.DATE_REQUESTED
